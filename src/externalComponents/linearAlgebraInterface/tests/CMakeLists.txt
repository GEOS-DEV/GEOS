	get_filename_component( test_name ${test} NAME_WE )

	                    SOURCES ${test}
	                    OUTPUT_DIR ${TEST_OUTPUT_DIRECTORY}
	                    DEPENDS_ON ${geosx_core_libs} gtest linearAlgebraInterface )

  blt_add_test( NAME ${test_name}
                COMMAND ${test_name}
                NUM_MPI_TASKS 2 )

endforeach()
