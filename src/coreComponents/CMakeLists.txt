
set( subdirs
     codingUtilities

set( coreLibLinkLine "" PARENT_SCOPE)
set( geosx_core_include_dirs "" )

foreach( lib ${subdirs} )
    add_subdirectory(${lib})
    if( NOT ( ${lib} STREQUAL "math" OR ${lib} STREQUAL "python" ) )
      list( APPEND coreLibs ${lib} )
    endif()
endforeach()
