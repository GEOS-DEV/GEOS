# Specify all headers

     common/layouts.hpp

     interfaces/blaslapack/BlasLapackFunctions.h
     interfaces/blaslapack/BlasLapackLA.hpp )

     interfaces/blaslapack/BlasLapackLA.cpp )

                 OBJECT     ${GEOS_BUILD_OBJ_LIBS} 
                 SHARED     ${GEOS_BUILD_SHARED_LIBS}
